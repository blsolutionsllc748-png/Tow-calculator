  <script>
    // Pricing configuration
    const BASE_FEE = 75;         // $75 base tow
    const PER_MILE = 3;          // $3 per mile
    const NON_RUNNER_FEE = 50;   // $50 extra if car doesn’t run

    let distanceService = null;

    function ensureDistanceServiceReady(callback, errorCallback) {
      if (distanceService) {
        callback();
        return;
      }
      if (window.google && google.maps && google.maps.DistanceMatrixService) {
        distanceService = new google.maps.DistanceMatrixService();
        callback();
      } else {
        errorCallback("Map service is not ready yet. Please wait a moment and try again.");
      }
    }

    window.addEventListener("load", function () {
      const btn = document.getElementById("calcBtn");
      const pickupInput = document.getElementById("pickup");
      const dropoffInput = document.getElementById("dropoff");
      const nonRunnerCheckbox = document.getElementById("nonRunner");
      const resultDiv = document.getElementById("result");

      btn.addEventListener("click", function () {
        const origin = pickupInput.value.trim();
        const destination = dropoffInput.value.trim();
        const isNonRunner = nonRunnerCheckbox.checked;

        if (!origin || !destination) {
          resultDiv.style.color = "red";
          resultDiv.textContent = "Please enter both pickup and drop-off addresses.";
          return;
        }

        resultDiv.style.color = "#111827";
        resultDiv.textContent = "Calculating distance and estimate...";

        ensureDistanceServiceReady(function () {
          distanceService.getDistanceMatrix(
            {
              origins: [origin],
              destinations: [destination],
              travelMode: google.maps.TravelMode.DRIVING,
              unitSystem: google.maps.UnitSystem.IMPERIAL
            },
            function (response, status) {
              if (status !== "OK") {
                resultDiv.style.color = "red";
                resultDiv.textContent = "Unable to calculate distance. Please check the addresses.";
                return;
              }

              const element = response.rows[0].elements[0];
              if (element.status !== "OK") {
                resultDiv.style.color = "red";
                resultDiv.textContent = "Route not found. Try different locations.";
                return;
              }

              const distanceMeters = element.distance.value;
              const distanceMiles = distanceMeters / 1609.34;
              const distanceText = element.distance.text;

              let total = BASE_FEE + (distanceMiles * PER_MILE);
              if (isNonRunner) {
                total += NON_RUNNER_FEE;
              }

              const totalFormatted = total.toFixed(2);
              const milesFormatted = distanceMiles.toFixed(1);

              resultDiv.style.color = "#111827";
              resultDiv.innerHTML =
                "<strong>Estimated towing cost: $" + totalFormatted + "</strong><br>" +
                "Distance: " + distanceText + " (" + milesFormatted + " miles)<br>" +
                "Final estimate may vary depending on route and vehicle condition.";
            }
          );
        }, function (msg) {
          resultDiv.style.color = "red";
          resultDiv.textContent = msg;
        });
      });
    });
  </script>
